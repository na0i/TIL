**학생**

```java
package cooperation;

public class Student {
    // 학생을 만들때 필요한 멤버 변수들
    String studentName;
    int grade;
    int money;
    
    // 학생을 만들때 생성자를 통해 이런 것을 받겠다
    public Student(String studentName, int money) {
        this.studentName = studentName;
        this.money = money
    }
    
    // 버스를 탄다는 method
    // 매개변수로 사용된 bus는 객체이므로 bus.을 입력하면 참조변수가 보인다.
    public void takeBus(Bus bus) {
        bus.take(1000); // bus의 method인 take호출 > bus에 1000원을 낸다.
        money -= 1000; // 학생의 돈은 감소
    }
    
    // 지하철을 탄다는 method
    public void takeSubway(Subway subway) {
        subway.take(1500); // bus의 method인 take호출 > bus에 1000원을 낸다.
        money -= 1000; // 학생의 돈은 감소
    }
    
    // 학생의 정보를 보여주는 method
    public void showInfo() {
        System.out.println(studentName + '님의 남은 돈은' + money + '입니다')
    }
}
```
